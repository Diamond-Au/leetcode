class Solution {
public:
  void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
    int k = m + n - 1; // 计算一下两个数组合并之后最后一个下标的位置
    int i = m - 1, j = n - 1;

    while(i>=0 && j >=0) {
      if(nums1[i] >= nums2[j]) nums1[k--] = nums1[i--];
      else nums1[k--] = nums2[j--];
    }
        // 只需遍历第二个数组就行 如果剩下的是第一个 那么他就应该在原来的位置上 不需要变动了
    while(j>=0) {
      nums1[k--] = nums2[j--];
    }
  }
};
